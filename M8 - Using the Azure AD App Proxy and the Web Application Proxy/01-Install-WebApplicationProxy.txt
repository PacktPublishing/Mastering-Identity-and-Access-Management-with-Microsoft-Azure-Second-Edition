gci Cert:\LocalMachine\My\ | fl
# Install Web Application Proxy
Install-WindowsFeature Web-Application-Proxy -IncludeManagementTools
# Configure Web Application Proxy
$creds = Get-Credential

# Server restart
Restart-Computer
